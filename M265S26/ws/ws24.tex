\item Give a rationale and/or proof for the following statements.
	\begin{enumerate}
	\item If $A$ is countably infinite and $B\subseteq A$, then $B$ is countable.
	\vfill
	\item If $A$ and $B$ are both countably infinite, then $A \cup B$ is countably infinite.
	\vfill
	\item If $A$ and $B$ are both countably infinite, then $A \times B$ is countably infinite.
	\vfill
\newpage
	\item If $A_1, A_2, \ldots A_k$ is a set of $k$ countably infinite sets, then
		\begin{enumerate}
		\item $A_1 \cup A_2 \cup \ldots \cup A_k$ is countably infinite.
		\vfill
		\item $A_1 \times A_2 \times \ldots \times A_k$ is countably infinite.
		\vfill
		\end{enumerate}
	\end{enumerate}

\item Match the statements on the left with an equivalent statement on the right for sets $A$ and $B$ and function $f:A \to B.$


\begin{tabular}{l c r}
$|A|=|B|$&\quad \quad \quad&$f$ is injective but not surjective\\
&&$f$ is surjective but not injective\\
$|A|\leq |B|$&&$f$ is surjective and injective\\
&&$f$ is neither surjective nor injective\\
$|A|< |B|$&&$f$ is injective \\
&&$f$ is surjective \\
\end{tabular}
\end{enumerate}
