 {\sc{Definition:}} A function is a rule that assigns to each $x$ (or ``input")  {\underline{\hspace{2in}} $y$ (or ``output"), called $f(x).$}\\
 
 {\sc{Example:}} Give an example of a rule whose inputs (or $x$'s) are PEOPLE and such that \\
 
 (1) the rule is a function; (2) the rule is NOT a function.\\

\hrulefill

{\sc{Evaluating a Function:}} Let $f(x)=x^2-3x.$ Find the indicated values.
\begin{enumerate}
\item $f(5)$
\vspace{.2in}
\item $f(-7)$
\vspace{.2in}
\item $f(2a)$
\vfill
\item $f(a+2)$
\vfill
\item $f(a)+2$
\vfill
\item $f(-x)$
\vfill
\item $f(\frac{a}{2})$
\vfill
\item $f(x^2)$
\vfill
\item $(f(x))^2$
\vfill
\end{enumerate}

{\sc{Evaluating a Function:}} Let $f(x) =
  \begin{cases} 
      \hfill 9 \hfill & \text{if $x \leq2$}\\
      \hfill \frac{x}{2}    \hfill & \text{ if $2 < x \leq 8$} \\
      \hfill 2x-1 \hfill & \text{ if $8<x$} \\
  \end{cases} $
  Find the indicated values.
  \begin{enumerate}
  \item $f(0)$
  \vspace{0.2in}
  \item $f(2)$
  \vspace{0.2in}
  \item $f(5)$
  \vspace{0.2in}
  \item $f(8)$
  \vspace{0.2in}
  \item $f(10)$
  \vspace{0.2in}
  \end{enumerate}
