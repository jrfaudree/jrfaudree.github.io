\item The Fundamental Theorem of Calculus (part 1):
\vfill
\item Find the derivative of each function below.
	\begin{multicols}{2}
	\begin{enumerate}
	\item $\d g(x)=\int_2^x (t^2-\tan (t) ) \: dt$
	\item $\d h(x)=\int_0^{\sin(x)} \sqrt{t^3+1} \: dt$
	\end{enumerate}
	\end{multicols}
\vfill

\vfill

	\end{enumerate}
