\begin{enumerate}
\item Two Versions of the Chain Rule
\vspace{2in}
\item Use version \fbox{B} to find $\frac{dy}{dx}$ if $y=3 \sqrt{u}$ and $u=\cos(x)+1.$\\
\vfill
\item For each function below, decompose the function into the form $y=f(u)$ and $u=g(x)$ and then find $\frac{dy}{dx}$ using version \fbox{B}.
	\begin{enumerate}
	\item $y=(3x-5)^8$
	\vfill
	\item $y=\displaystyle{\large{\frac{1}{x^3+\tan(x)}}}$
	\vfill
	\end{enumerate}
	\newpage
	
\item Find $\frac{dy}{dx}$ using version $\fbox{A}.$
	\begin{enumerate}
	\item $y=(\frac{1}{x^2}+\frac{x^2}{3})^4$
	\vfill
	\item $y= \cos(2x)$
	\vfill
	\item $y=\sqrt{x^2+\sin(x)}$
	\vfill
	\item $y=x\tan(\frac{\pi x}{4})$
	\vfill
	\item $y=\frac{x}{\sin^2(x)}$
	\vfill
	\end{enumerate}
\end{enumerate}
